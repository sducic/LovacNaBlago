create or replace TRIGGER "LEGENDA_AUTO_PK" 
    BEFORE INSERT
    ON LEGENDA
    REFERENCING NEW AS NEW
    FOR EACH ROW
BEGIN
    SELECT LEGENDA_IDLEG_SEQ.NEXTVAL INTO :NEW.IDLEG FROM DUAL;
END;
